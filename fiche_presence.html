function generatePresencePdf() { const { jsPDF } = window.jspdf; const pdf = new jsPDF("landscape", "mm", "a4"); const monthLabel = formatMonthLabel(state.presence.month); const relayMonthLabel = formatMonthLabel(state.relay.month); const leftMargin = 8; const topMargin = 10; const dayW = 7.2; const rowH = 8; const nameW = 36; const totalW = 12; // nombre réel de jours du mois sélectionné const pdfDaysInPresenceMonth = getDaysInMonth(state.presence.month); const pdfDaysInRelayMonth = getDaysInMonth(state.relay.month); pdf.setFont("helvetica", "bold"); pdf.setFontSize(12); pdf.text("FEUILLE DE PRESENCE AU DOMICILE DE L'ASSISTANT FAMILIAL", leftMargin, topMargin); pdf.setFont("helvetica", "normal"); pdf.setFontSize(9); pdf.text(`Nom et prénom de l'assistant familial : ${state.presence.afName || ""}`, leftMargin, 18); pdf.text(`Mois : ${monthLabel}`, 220, 18); let startX = leftMargin; let startY = 25; drawPdfCell(pdf, startX, startY, nameW, rowH, "ENFANT", "center", 7); for (let d = 1; d <= pdfDaysInPresenceMonth; d += 1) { drawPdfCell(pdf, startX + nameW + (d - 1) * dayW, startY, dayW, rowH, d, "center", 6); } drawPdfCell( pdf, startX + nameW + pdfDaysInPresenceMonth * dayW, startY, totalW, rowH, "Total", "center", 7 ); for (let row = 0; row < state.presence.children.length; row += 1) { const y = startY + rowH * (row + 1); const childName = state.presence.children[row] || `Enfant ${row + 1}`; drawPdfCell(pdf, startX, y, nameW, rowH, childName, "left", 7); let total = 0; for (let d = 0; d < pdfDaysInPresenceMonth; d += 1) { const code = state.presence.grid[row][d] || ""; if (code === "P") total += 1; drawPdfCell(pdf, startX + nameW + d * dayW, y, dayW, rowH, code, "center", 6); } drawPdfCell( pdf, startX + nameW + pdfDaysInPresenceMonth * dayW, y, totalW, rowH, total, "center", 7 ); } let infoY = startY + rowH * (state.presence.children.length + 2) + 6; pdf.setFont("helvetica", "bold"); pdf.setFontSize(9); pdf.text("Rappel motifs :", leftMargin, infoY); pdf.setFont("helvetica", "normal"); pdf.text( "P = Présence I = Internat DVH = Droits de visite / hébergement A = Absence F = Fugue R = Relais", leftMargin + 22, infoY ); infoY += 10; pdf.setFont("helvetica", "bold"); pdf.text("Rectificatif :", leftMargin, infoY); pdf.setFont("helvetica", "normal"); pdf.rect(leftMargin, infoY + 2, 180, 20); pdf.text(state.presence.rectif || "", leftMargin + 2, infoY + 8, { maxWidth: 176 }); pdf.text(`Date : ${new Date().toLocaleDateString("fr-FR")}`, 220, infoY + 8); pdf.text("Signature :", 220, infoY + 18); pdf.rect(220, infoY + 20, 55, 18); addSignatureToPdf(pdf, state.presence.signatureDataUrl, 220, infoY + 20, 55, 18); pdf.addPage(); pdf.setFont("helvetica", "bold"); pdf.setFontSize(12); pdf.text("FICHE DE RELAIS DU MOIS PRECEDENT", leftMargin, topMargin); pdf.setFont("helvetica", "normal"); pdf.setFontSize(9); pdf.text(`Nom et prénom de l'assistant familial signataire : ${state.relay.afName || ""}`, leftMargin, 18); pdf.text(`Mois : ${relayMonthLabel}`, 220, 18); const relayNameW = 34; const relayAfW = 36; const relayDayW = 5.8; const relayTotalW = 12; let relayY = 25; drawPdfCell(pdf, leftMargin, relayY, relayNameW, rowH, "ENFANT", "center", 7); drawPdfCell(pdf, leftMargin + relayNameW, relayY, relayAfW, rowH, "AF habituel", "center", 7); for (let d = 1; d <= pdfDaysInRelayMonth; d += 1) { drawPdfCell( pdf, leftMargin + relayNameW + relayAfW + (d - 1) * relayDayW, relayY, relayDayW, rowH, d, "center", 6 ); } drawPdfCell( pdf, leftMargin + relayNameW + relayAfW + pdfDaysInRelayMonth * relayDayW, relayY, relayTotalW, rowH, "Total", "center", 7 ); state.relay.rows.forEach((row, index) => { const y = relayY + rowH * (index + 1); let total = 0; drawPdfCell(pdf, leftMargin, y, relayNameW, rowH, row.child || "", "left", 7); drawPdfCell(pdf, leftMargin + relayNameW, y, relayAfW, rowH, row.usualAf || "", "left", 7); for (let d = 0; d < pdfDaysInRelayMonth; d += 1) { const code = row.days[d] || ""; if (code === "P") total += 1; drawPdfCell( pdf, leftMargin + relayNameW + relayAfW + d * relayDayW, y, relayDayW, rowH, code, "center", 6 ); } drawPdfCell( pdf, leftMargin + relayNameW + relayAfW + pdfDaysInRelayMonth * relayDayW, y, relayTotalW, rowH, total, "center", 7 ); }); let relayBottomY = relayY + rowH * (state.relay.rows.length + 2) + 10; pdf.setFont("helvetica", "bold"); pdf.text("Remarque :", leftMargin, relayBottomY); pdf.setFont("helvetica", "normal"); pdf.rect(leftMargin, relayBottomY + 2, 180, 18); pdf.text(state.relay.notes || "", leftMargin + 2, relayBottomY + 8, { maxWidth: 176 }); pdf.text(`Date : ${new Date().toLocaleDateString("fr-FR")}`, 220, relayBottomY + 8); pdf.text("Signature :", 220, relayBottomY + 18); pdf.rect(220, relayBottomY + 20, 55, 18); addSignatureToPdf(pdf, state.presence.signatureDataUrl, 220, relayBottomY + 20, 55, 18); const fileName = `fiche_presence_${state.presence.month || "sans_mois"}.pdf`; const pdfBlob = pdf.output("blob"); addPdfToGlobalHistory(pdfBlob, fileName, monthLabel); pdf.save(fileName); showToast("PDF généré et ajouté à l’historique"); }
